146  static inline InstructionPacket<protocol_t> set_goal_position_angle(typename Servo<Model>::protocol_t::id_t id, double rad)
147  {
148  double deg = rad * 57.2958;
149  if (!(deg >= ct_t::min_goal_angle_deg && deg <= ct_t::max_goal_angle_deg))
150  throw errors::ServoLimitError(id, ct_t::min_goal_angle_deg, ct_t::max_goal_angle_deg, deg);
151  typename ct_t::goal_position_t pos = ((deg - ct_t::min_goal_angle_deg) * (ct_t::max_goal_position - ct_t::min_goal_position) / (ct_t::max_goal_angle_deg - ct_t::min_goal_angle_deg)) + ct_t::min_goal_position;
152  return set_goal_position(id, pos);

237  {
238  typename Servo<Model>::ct_t::present_speed_t speed;
239  Servo<Model>::protocol_t::unpack_data(st.parameters(), speed);
240 
241  int8_t sign;
242  if (Servo<Model>::ct_t::speed_sign_bit) { // the highest bit is used for the sign
243  sign = speed / Servo<Model>::ct_t::max_goal_speed == 0 ? 1 : -1;
244  speed = speed % Servo<Model>::ct_t::max_goal_speed;
245  }
246  else {
247  sign = 1;
248  }
249 
250  double speed_rpm = sign * speed * Servo<Model>::ct_t::rpm_per_tick;
251  double speed_si = speed_rpm * 0.104720; // convertion ratio is 2*pi/60
252  return speed_si;
253  }
